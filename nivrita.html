    <script>

        // Data copied from the Python file for analysis

        const AMINO_ACID_PROPERTIES = {

            'A': {'name': 'Alanine', 'mass': 89.09, 'hydrophobicity': 1.8},

            'R': {'name': 'Arginine', 'mass': 174.20, 'hydrophobicity': -4.5},

            'N': {'name': 'Asparagine', 'mass': 132.12, 'hydrophobicity': -3.5},

            'D': {'name': 'Aspartic acid', 'mass': 133.10, 'hydrophobicity': -3.5},

            'C': {'name': 'Cysteine', 'mass': 121.16, 'hydrophobicity': 2.5},

            'Q': {'name': 'Glutamine', 'mass': 146.15, 'hydrophobicity': -3.5},

            'E': {'name': 'Glutamic acid', 'mass': 147.13, 'hydrophobicity': -3.5},

            'G': {'name': 'Glycine', 'mass': 75.07, 'hydrophobicity': -0.4},

            'H': {'name': 'Histidine', 'mass': 155.16, 'hydrophobicity': -3.2},

            'I': {'name': 'Isoleucine', 'mass': 131.18, 'hydrophobicity': 4.5},

            'L': {'name': 'Leucine', 'mass': 131.18, 'hydrophobicity': 3.8},

            'K': {'name': 'Lysine', 'mass': 146.19, 'hydrophobicity': -3.9},

            'M': {'name': 'Methionine', 'mass': 149.21, 'hydrophobicity': 1.9},

            'F': {'name': 'Phenylalanine', 'mass': 165.19, 'hydrophobicity': 2.8},

            'P': {'name': 'Proline', 'mass': 115.13, 'hydrophobicity': -1.6},

            'S': {'name': 'Serine', 'mass': 105.09, 'hydrophobicity': -0.8},

            'T': {'name': 'Threonine', 'mass': 119.12, 'hydrophobicity': -0.7},

            'W': {'name': 'Tryptophan', 'mass': 204.23, 'hydrophobicity': -0.9},

            'Y': {'name': 'Tyrosine', 'mass': 181.19, 'hydrophobicity': -1.3},

            'V': {'name': 'Valine', 'mass': 117.15, 'hydrophobicity': 4.2},

        };


        const WATER_LOSS_MASS = 18.015; // Mass of water lost per peptide bond


        /**

         * Cleans the sequence and performs the analysis, updating the UI.

         */

        function analyzeSequence() {

            const sequenceInput = document.getElementById('sequenceInput').value;

            const errorMsg = document.getElementById('errorMsg');

            const resultsArea = document.getElementById('resultsArea');

            

            // 1. Prepare and validate sequence

            const sequence = sequenceInput.replace(/[^a-zA-Z]/g, '').toUpperCase();

            

            errorMsg.classList.add('hidden');

            resultsArea.classList.add('hidden');


            if (sequence.length === 0) {

                errorMsg.textContent = 'Please enter a valid protein sequence.';

                errorMsg.classList.remove('hidden');

                return;

            }


            // 2. Core Analysis Logic

            const totalLength = sequence.length;

            let totalMass = 0;

            let totalHydrophobicity = 0;

            const counts = {};


            for (const aa of sequence) {

                if (AMINO_ACID_PROPERTIES[aa]) {

                    counts[aa] = (counts[aa] || 0) + 1;

                    totalMass += AMINO_ACID_PROPERTIES[aa].mass;

                    totalHydrophobicity += AMINO_ACID_PROPERTIES[aa].hydrophobicity;

                } else {

                    // Alert user to non-standard AA codes, but continue with valid ones

                    console.warn(`Skipping non-standard amino acid code: ${aa}`);

                }

            }


            if (Object.keys(counts).length === 0) {

                 errorMsg.textContent = 'The sequence only contained non-standard characters.';

                 errorMsg.classList.remove('hidden');

                 return;

            }


            // Calculate final properties

            const waterLossMass = WATER_LOSS_MASS * (totalLength - 1);

            const estimatedMolecularWeight = totalMass - waterLossMass;

            const averageHydrophobicity = totalHydrophobicity / totalLength;


            // Calculate composition

            const composition = [];

            for (const aa in counts) {

                const percent = (counts[aa] / totalLength) * 100;

                composition.push({

                    aa: aa,

                    name: AMINO_ACID_PROPERTIES[aa].name,

                    percent: percent

                });

            }


            // Sort composition by percentage (highest first)

            composition.sort((a, b) => b.percent - a.percent);


            // 3. Update the UI

            document.getElementById('resultLength').textContent = totalLength.toLocaleString();

            document.getElementById('resultMass').textContent = estimatedMolecularWeight.toFixed(2);

            document.getElementById('resultHydrophobicity').textContent = averageHydrophobicity.toFixed(3);

            

            const interpretation = averageHydrophobicity > 0 ? 

                `The average hydrophobicity index is positive (${averageHydrophobicity.toFixed(3)}), suggesting this protein segment is more likely to be found in a hydrophobic (water-repelling) environment, like a cell membrane.` :

                `The average hydrophobicity index is negative (${averageHydrophobicity.toFixed(3)}), suggesting this protein segment is more likely to be found in a hydrophilic (water-attracting) environment, like cytoplasm.`;


            document.getElementById('resultInterpretation').textContent = interpretation;


            const compositionList = document.getElementById('compositionList');

            compositionList.innerHTML = ''; // Clear old results


            composition.forEach(item => {

                const div = document.createElement('div');

                div.className = 'p-3 bg-gray-50 border border-gray-200 rounded-lg';

                div.innerHTML = `

                    <p class="text-sm font-bold text-gray-900"><span class="matrix bg-gray-200 rounded px-1 py-0.5">${item.aa}</span> ${item.name}</p>

                    <p class="text-xl font-semibold text-purple-700 mt-1">${item.percent.toFixed(2)}%</p>

                `;

                compositionList.appendChild(div);

            });


            resultsArea.classList.remove('hidden');

        }


        // Run analysis on load using the default example sequence

        document.addEventListener('DOMContentLoaded', analyzeSequence);

    </script>
